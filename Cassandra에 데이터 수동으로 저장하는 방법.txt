# csv파일 Cassandra로 옮기기
kubectl -n traininghost cp "{데이터셋 경로}" cassandra-0:/tmp/{데이터셋 파일명}

# Cassandra 실행
kubectl exec -it {cassandra 파드명} -n traininghost -- bash

# Cassandra 내 csv 파일이 존재하는지 확인
I have no name!@cassandra-0:/tmp$ head -n 3 /tmp/{데이터셋 파일명}

# Cassandra cqlsh 모드 접속
I have no name!@cassandra-0:/$ cqlsh cassandra.traininghost.svc.cluster.local 9042 -u cassandra -p {cassandra password}

# Cassandra 테이블 생성
cassandra@cqlsh: DESC keyspaces		-> feature_store가 존재하는지 확인
cassandra@cqlsh: USE feature_store;
cassandra@cqlsh: DESC tables;		-> 테이블 목록 조회
cassandra@cqlsh:feature_store> CREATE TABLE IF NOT EXISTS {테이블명*규칙 잘 지켜서 넣어줄 것} (
traffic_type        int,  
src_port            float,
dst_port            float,
duration            float,
src_bytes           float,
dst_bytes           float,
missed_bytes        float,
src_pkts            float,
src_ip_bytes        float,
dst_pkts            float,
dst_ip_bytes        float,
ip_proto            float,
http_trans_depth    float,
files_total_bytes   float,
is_GET_mthd         float,
http_status_error   float,
is_file_transfered  float,
idx                 int,
PRIMARY KEY(idx)
);

# table 생성 확인
cassandra@cqlsh:feature_store> SELECT * FROM {테이블명};

# 접속 끊고 다시 cqlsh 모드로 접속
cassandra@cqlsh:feature_store> EXIT;
I have no name!@cassandra-0:/$ cqlsh cassandra.traininghost.svc.cluster.local 9042 -u cassandra -p {cassandra password}

# 데이터셋 파일을 table로 복사
cassandra@cqlsh: COPY feature_store.{테이블명} FROM '/tmp/{데이터셋 파일명}';

# 데이터셋 파일의 실제 데이터를 테이블에 저장
cassandra@cqlsh: USE feature_store; 
cassandra@cqlsh:feature_store> COPY feature_store.{테이블명} (
traffic_type, src_port, dst_port, duration, src_bytes, dst_bytes, missed_bytes, src_pkts, src_ip_bytes, dst_pkts, dst_ip_bytes, ip_proto, http_trans_depth, files_total_bytes, is_get_mthd, http_status_error, is_file_transfered, idx)
FROM '/tmp/{데이터셋 파일명}'
WITH HEADER=TRUE AND DELIMITER=',' AND NULL='' AND NUMPROCESSES=1;
